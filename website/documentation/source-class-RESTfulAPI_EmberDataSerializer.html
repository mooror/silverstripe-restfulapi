 26:   /**
 27:    * Sideloaded records settings
 28:    * Specify which relation's records will be added to the JSON root:
 29:    * 'RequestedClass' => array('RelationNameToSideLoad', 'Another')
 30:    * 
 31:    * Non sideloaded response:
 32:    * {
 33:    *   'member': {
 34:    *     'name': 'John',
 35:    *     'favourites': [1, 2]
 36:    *   }
 37:    * }
 38:    *
 39:    * Response with sideloaded records:
 40:    * {
 41:    *   'member': {
 42:    *     'name': 'John',
 43:    *     'favourites': [1, 2]
 44:    *    },
 45:    *
 46:    *        favourites': [{
 47:    *       'id': 1,
 48:    *       'name': 'Mark'
 49:    *    },{
 50:    *       'id': 2,
 51:    *       'name': 'Maggie'
 52:    *    }]
 53:    * }
 54:    *
 55:    * Try not to use in conjunction with {@link RESTfulAPI} $embedded_records
 56:    * with the same settings.
 57:    * 
 58:    * @var array
 59:    * @config
 60:    */
 61:   private static $sideloaded_records;

 69:   protected $sideloadedRecords;

 73:    * Construct and set current config

 75:   public function __construct()
 76:   {
 77:       parent::__construct();

 79:       $sideloaded_records = Config::inst()->get('RESTfulAPI_EmberDataSerializer', 'sideloaded_records');
 80:       if (is_array($sideloaded_records)) {
 81:           $this->sideloadedRecords = $sideloaded_records;
 82:       } else {

 88:     /**
 89:      * Convert raw data (DataObject or DataList) to JSON
 90:      * ready to be consumed by the client API
 91:      * 
 92:      * @param  DataObject|DataList  $data  Data to serialize
 93:      * @return string                      JSON representation of data
 94:      */
 95:     public function serialize($data)
 96:     {
 97:         $json = '';
 98:         $formattedData = null;
 99: 
100:         if ($data instanceof DataObject) {
101:             $dataClass     = $data->ClassName;
102:             $rootClassName = $this->formatName($dataClass);
103:             $formattedData = $this->formatDataObject($data);
104:         } elseif ($data instanceof DataList) {
105:             $dataClass     = $data->dataClass;
106:             $rootClassName = $this->formatName($data->dataClass);
107:             $rootClassName = Inflector::pluralize($rootClassName);
108:             $formattedData = $this->formatDataList($data);
109:         }
110: 
111:         if ($formattedData !== null) {
112:             $root = new stdClass();
113:             $root->{$rootClassName} = $formattedData;
114: 
115:         // if it's not an empty response >> check if we should be sideloading some data
116:         if ($formattedData && $this->hasSideloadedRecords($dataClass)) {
117:             $root = $this->insertSideloadData($root, $data);
118:         }
119: 
120:             $json = $this->jsonify($root);
121:         } else {
122:             //fallback: convert non array to object then encode
123:             if (!is_array($data)) {
124:                 $data = (object) $data;
125:             }
126:             $json = $this->jsonify($data);
127:         }
128: 
129:         return $json;
130:     }

134:      * Format a SilverStripe ClassName or Field name

137:      * @param  string $name ClassName of DBField name
138:      * @return string       Formatted name
139:      */
140:     public function formatName($name)
141:     {
142:         $class = Inflector::singularize($name);
143: 
144:         if (ClassInfo::exists($class)) {
145:             $name = lcfirst($class);
146:         } else {
147:             $name = $this->serializeColumnName($name);
148:         }

173:     /**
174:      * Check if a specific class requires data to be sideloaded.
175:      * 
176:      * @param  string  $classname Requested data classname
177:      * @return boolean            True if some relations should be sideloaded
178:      */
179:     protected function hasSideloadedRecords($classname)
180:     {
181:         return array_key_exists($classname, $this->sideloadedRecords);
182:     }
183: 
184: 
185:     /**
186:      * Fetches and return all the data that need to be sideloaded
187:      * for a specific source DataObject or DataList.
188:      * 
189:      * @param  DataObject|DataList  $dataSource The source data to fetch sideloaded records for
190:      * @return array                            A map of relation names with their data
191:      */
192:     protected function getSideloadData($dataSource)
193:     {
194:         $data = array();
195: 
196:         if ($dataSource instanceof DataObject) {
197:             $has_one           = $dataSource->stat('has_one');
198:             $has_many          = $dataSource->stat('has_many');
199:             $many_many         = $dataSource->stat('many_many');
200:             $belongs_many_many = $dataSource->stat('belongs_many_many');
201: 
202:             $relationsMap = array();
203:             if (is_array($has_one)) {
204:                 $relationsMap = array_merge($relationsMap, $has_one);
205:             }
206:             if (is_array($has_many)) {
207:                 $relationsMap = array_merge($relationsMap, $has_many);
208:             }
209:             if (is_array($many_many)) {
210:                 $relationsMap = array_merge($relationsMap, $many_many);
211:             }
212:             if (is_array($belongs_many_many)) {
213:                 $relationsMap = array_merge($relationsMap, $belongs_many_many);
214:             }
215: 
216:             // if a single DataObject get the data for each relation
217:             foreach ($this->sideloadedRecords[$dataSource->ClassName] as $relationName) {
218:                 $newData = $this->getEmbedData($dataSource, $relationName);
219: 
220:                 if ($newData !== null) {
221:                     // has_one are only simple array and we want arrays or array
222:           if (is_array($has_one) && in_array($relationName, $has_one)) {
223:               $newData = array($newData);
224:           }
225:           
226:                     $relationClass        = $relationsMap[$relationName];
227:                     $data[$relationClass] = $newData;
228:                 }
229:             }
230:         } elseif ($dataSource instanceof DataList) {
231:             // if a list of DataObject, loop through each and merge all the data together
232:             foreach ($dataSource as $dataObjectSource) {
233:                 $sideloadData = $this->getSideloadData($dataObjectSource);
234:                 
235:                 //combine sideloaded records
236:                 foreach ($sideloadData as $class => $extraData) {
237:                     if (!isset($data[$class])) {
238:                         $data[$class] = array();
239:                     }
240: 
241:                     // merge for list of records, push for individual record
242:                     $arrayOfArrays = array_filter($extraData, 'is_array');
243:                     if (count($arrayOfArrays) == count($extraData)) {
244:                         //print_r('array_merge');
245:                         $data[$class] = array_merge($data[$class], $extraData);
246:                     } else {
247:                         //print_r('array_push');
248:                         array_push($data[$class], $extraData);
249:                     }
250:                 }
251:             }
252: 
253:             // remove duplicates
254:             foreach ($data as $relationClass => $relationData) {
255:                 $data[$relationClass] = array_unique($relationData, SORT_REGULAR);
256:             }
257:         }
258:         
259:         return $data;
260:     }
261: 
262: 
263:     /**
264:      * Take a root object ready to be converted into JSON
265:      * and an original data source (DataObject OR DataList)
266:      * and insorts into the root object all relation records
267:      * that should be sideloaded.
268:      * 
269:      * @param  stdClass              $root       Root object ready to become JSON
270:      * @param  DataObject|DataList   $dataSource The original data set from the root object
271:      * @return stdClass                          The updated root object sith the sideloaded data attached
272:      */
273:     protected function insertSideloadData(stdClass $root, $dataSource)
274:     {
275:         // get the extra data
276:     $sideloadData = $this->getSideloadData($dataSource);
277: 
278:     // attached those to the root
279:     foreach ($sideloadData as $relationClass => $relationData) {
280:         $rootRelationClass = $this->formatName($relationClass);
281: 
282:         // pluralize only set of records
283:         $allArrays = array_filter($relationData, 'is_array');
284:         if (count($allArrays) == count($relationData)) {
285:             $rootRelationClass = Inflector::pluralize($rootRelationClass);
286:         }
287: 
288:             // attach to root
289:             $root->{$rootRelationClass} = $relationData;
290:     }
291: 
292:         return $root;
293:     }
294: }
